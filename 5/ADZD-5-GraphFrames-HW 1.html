def send_fake(graph, steps, spread_factor, start_node):
    # init post
    g = graph
    is_post = g.vertices.withColumn("is_post", lit(0))
    g = GraphFrame(is_post, g.edges)

    new_pages = []
    start_page_id = str(start_node)
    df_spark = g.vertices
    first_post = df_spark.withColumn('is_post', 
                                   F.when(df_spark['id'] == start_page_id, 1)
                                    .otherwise(df_spark['is_post']))
    print(f"Step 0: 1 pages published")
    new_pages.append(1)
    g = GraphFrame(first_post, g.edges)
    
    for step in range(steps):
        # send massage to neighbours
        messages = g.edges.join(g.vertices, g.edges.src == g.vertices.id) \
            .where(col("is_post") == 1) \
            .withColumn("spread", (rand() < spread_factor).cast("int")) \
            .select(col("dst").alias("id"), "spread")
    
        # update status "is_post" based on massages
        updated = messages.groupBy("id").agg({"spread": "max"}) \
            .withColumnRenamed("max(spread)", "spread") \
            .join(g.vertices, "id", "right_outer") \
            .fillna(0, ["spread"]) \
            .withColumn("is_post", when((col("is_post") == 1) | (col("spread") == 1), 1).otherwise(0)) \
            .select("id", "facebook_id", "page_name", "page_type", "is_post")
    
        g = GraphFrame(updated, g.edges)
    
        # # number of posted sites
        count_published = g.vertices.filter(col("is_post") == 1).count() - sum(new_pages)
        new_pages.append(count_published)
        print(f"Step {step + 1}: {count_published} pages published")
    return new_pages

def plot_pages_steps(graph, steps=5, spread_factor=0.25, start_node=1):
    pages = send_fake(graph, steps=steps, spread_factor=spread_factor, start_node=start_node)
    plt.scatter(x=list(range(1, len(pages) + 1)) , y=pages)
    plt.xlabel('steps')
    plt.ylabel('posted pages')
    plt.title(f'relationship of posted pages to steps for starting node {start_node} and spread factor={spread_factor}')
    plt.show()
